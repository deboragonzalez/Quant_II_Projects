# Flexibility Variables: creating a sub-dataset of the weekly dataset averaging flexibility measures to then join to our main dataset
flexibility_avg_kff_data <- weekly_kff_data %>%
  group_by(ID) %>%
  summarise(
    # if all NAs for one ID, then avg variable will be NA
    avg_Sess_Fit = if (all(is.na(Sess_Fit))) NA_real_ else mean(Sess_Fit, na.rm = TRUE), # na.rm = TRUE is telling R to ignore NA values for now
    avg_Sess_Tail = if (all(is.na(Sess_Tail))) NA_real_ else mean(Sess_Tail, na.rm = TRUE),
    .groups = "drop" # ungrouping for future ease
  )

# Doing the same for SD variables

# Note on latent variable scale interpretation: they are factor scores estimated by lavPredict(), not observed raw scores. Each latent variable was scaled by fixing its variance to 1 in the model, instead of fixing the first loading to 1. Latent means are fixed to 0 by default in a single-group model.So, A score like 0 means the person is around the sample’s latent mean, +1 means about one latent standard deviation above the sample mean, -1 means about one latent standard deviation below the sample mean. This means that higher scores indicate more of the latent construct, scores are centered approximately around zero, and a 1-unit difference is roughly a 1-SD difference on the latent factor scale.
